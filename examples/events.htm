<script type="text/javascript"> 
        function show(stat) {
            switch (stat) {
                case 1: $('#result').text('"beforeAppear" event was fired.');
                    $('#result').css('backgroundColor', '#ffb8b8');
                    break;
                case 2: $('#result').text('"appearFlashCompleted" event was fired.');
                    $('#result').css('backgroundColor', '#bfffc5');
                    break;
                case 3: $('#result').text('"appearCompleted" event was fired.');
                    $('#result').css('backgroundColor', '#fabfff');
                    break;
                case 4: $('#result').text('"beforeDisappear" event was fired.');
                    $('#result').css('backgroundColor', '#ffc97c');
                    break;
                case 5: $('#result').text('"disappearFlashCompleted" event was fired.');
                    $('#result').css('backgroundColor', '#bff7ff');
                    break;
                case 6: $('#result').text('"disappearCompleted" event was fired.');
                    $('#result').css('backgroundColor', '#f7ffbf');
                    break;
            }
        }
$('#exam').tooltip( 'Three beautiful Penguins' , { appearSpeed: 4000, appearFlashInterval: 4000, disappearSpeed: 10000, disappearFlashInterval: 2500, beforeAppear: function () { show(1); }, appearFlashCompleted: function () { show(2); }, appearCompleted: function () { show(3); }, beforeDisappear: function () { show(4); }, disappearFlashCompleted: function () { show(5); }, disappearCompleted: function () { show(6); } } );
</script>
